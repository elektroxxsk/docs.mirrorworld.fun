---
title: Example Usage of Mirror World SDK
---

```jsx
import { useState } from "react";
import { MirrorWorld } from "@mirrorworld/web3.js"

export function App() {
  // 1. Initialize Mirror World SDK
  const mirrorworld = new MirrorWorld({
    apiKey: "SECRET_API_KEY",
  })

  const [user, setUser] = useState()

  // 2. Login user with Social Auth Provider
  async function loginWithGoogle() {
    const user = await mirrorworld.login()
    setUser(user)
  }

  return (
    <div>
      Hello World
      <button onClick={loginWithGoogle}>
        Login
      </button>

      {/* 3. Display user wallet address if logged in */}
      {user && (
        <div>
          Welcome, {user.wallet.sol_address}!
        </div>
      )}
    </div>
  )
}
```
